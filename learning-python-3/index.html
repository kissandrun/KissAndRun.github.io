def learnarg(*args, **kwargs):
    for i in args + tuple(kwargs.values()):
        print(i)

    for key, value in kwargs.items():
        print('{}={}'.format(key, value))

learnarg('Wo', 4, 'KissAndRun', b=7, a=5, c='Hi')
